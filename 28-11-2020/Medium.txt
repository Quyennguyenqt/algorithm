https://leetcode.com/problems/widest-vertical-area-between-two-points-containing-no-points/
class Solution {
    	public static void maxWidthOfVerticalArea(int[][] point) {
		int n[] = new int[point.length];
		int lent = n.length;
		for (int i = 0; i < lent; i++) {
			n[i] = point[i][0];
		}
		Arrays.sort(n);
		int max = 0;
		for (int i = 0; i < n.length - 1; i++) {
			if (n[i + 1] - n[i] > max) {
				max = n[i + 1] - n[i];
			}
		}
		return max;
	}
}
